### Step 2: Add `create_repository()` to RepositoryMixin

Add to `solune/backend/src/services/github_projects/repository.py`:

```python
async def create_repository(
    self,
    access_token: str,
    owner: str,
    name: str,
    *,
    private: bool = True,
    description: str = "",
    is_org: bool = False,
) -> dict:
    github = await self._get_client(access_token)
    if is_org:
        response = await github.rest.repos.async_create_in_org(
            org=owner, name=name, private=private,
            auto_init=True, description=description,
        )
    else:
        response = await github.rest.repos.async_create_for_authenticated_user(
            name=name, private=private,
            auto_init=True, description=description,
        )
    repo = response.parsed_data
    return {
        "id": repo.id,
        "node_id": repo.node_id,
        "name": repo.name,
        "full_name": repo.full_name,
        "html_url": str(repo.html_url),
        "default_branch": repo.default_branch,
    }
```

### Step 3: Add `list_available_owners()` to RepositoryMixin

```python
async def list_available_owners(self, access_token: str) -> list[dict]:
    github = await self._get_client(access_token)
    user_resp = await github.rest.users.async_get_authenticated()
    user = user_resp.parsed_data
    owners = [{"login": user.login, "avatar_url": str(user.avatar_url), "type": "User"}]

    orgs_resp = await github.rest.orgs.async_list_for_authenticated_user()
    for org in orgs_resp.parsed_data:
        owners.append({"login": org.login, "avatar_url": str(org.avatar_url), "type": "Organization"})
    return owners
```

### Step 4: Add `create_project_v2()` and `link_project_to_repository()` to ProjectsMixin

Add to `solune/backend/src/services/github_projects/projects.py`:

```python
async def create_project_v2(self, access_token: str, owner: str, title: str, *, is_org: bool = False) -> dict:
    # Get owner node_id
    github = await self._get_client(access_token)
    if is_org:
        resp = await github.rest.orgs.async_get(org=owner)
        owner_node_id = resp.parsed_data.node_id
    else:
        resp = await github.rest.users.async_get_by_username(username=owner)
        owner_node_id = resp.parsed_data.node_id

    # Create project
    data = await self._graphql(access_token, CREATE_PROJECT_V2_MUTATION, {
        "ownerId": owner_node_id, "title": title
    })
    project = data["createProjectV2"]["projectV2"]

    # Best-effort: configure status columns
    try:
        await self._configure_project_status(access_token, project["id"])
    except Exception as exc:
        logger.warning("Failed to configure status columns for project %s: %s", project["id"], exc)

    return {"id": project["id"], "number": project["number"], "url": project["url"]}

async def link_project_to_repository(self, access_token: str, project_id: str, repository_id: str) -> None:
    await self._graphql(access_token, LINK_PROJECT_V2_TO_REPO_MUTATION, {
        "projectId": project_id, "repositoryId": repository_id
    })
```

```python
import os
from src.config import get_settings

GENERIC_COPILOT_INSTRUCTIONS = """# Copilot Instructions
> Auto-generated by Solune. Customize for your project.

## Tech Stack
<!-- TODO: List your primary languages, frameworks, and tools -->

## Coding Conventions
<!-- TODO: Describe naming conventions, code style, and patterns -->

## Testing
<!-- TODO: Describe testing approach, frameworks, and coverage -->

## Deployment
<!-- TODO: Describe deployment process, environments, and CI/CD -->
"""

_template_cache: list[dict[str, str]] | None = None

async def build_template_files(repo_name: str, display_name: str) -> list[dict[str, str]]:
    global _template_cache
    if _template_cache is None:
        _template_cache = _load_templates()
    return list(_template_cache)  # Return a copy

def _load_templates() -> list[dict[str, str]]:
    settings = get_settings()
    source_dir = getattr(settings, 'template_source_dir', None) or os.getcwd()
    files = []
    for subdir in ['.github', '.specify']:
        _walk_dir(source_dir, subdir, files)
    gitignore = os.path.join(source_dir, '.gitignore')
    if os.path.isfile(gitignore):
        with open(gitignore) as f:
            files.append({"path": ".gitignore", "content": f.read()})
    return files
```

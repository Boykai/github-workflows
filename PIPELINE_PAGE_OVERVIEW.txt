## 5. PIPELINE API - /solune/frontend/src/services/api.ts

COMPLETE PIPELINESAPI OBJECT:

export const pipelinesApi = {
  // LIST OPERATIONS
  list(projectId: string, sort?: string, order?: string)
    → Promise<PipelineConfigListResponse>
    GET /pipelines/{projectId}

  // SINGLE ITEM OPERATIONS
  get(projectId: string, pipelineId: string)
    → Promise<PipelineConfig>
    GET /pipelines/{projectId}/{pipelineId}

  // CREATE OPERATIONS
  create(projectId: string, data: PipelineConfigCreate)
    → Promise<PipelineConfig>
    POST /pipelines/{projectId}

  // UPDATE OPERATIONS
  update(projectId: string, pipelineId: string, data: PipelineConfigUpdate)
    → Promise<PipelineConfig>
    PATCH /pipelines/{projectId}/{pipelineId}

  // DELETE OPERATIONS
  delete(projectId: string, pipelineId: string)
    → Promise<{ success: boolean; deleted_id: string }>
    DELETE /pipelines/{projectId}/{pipelineId}

  // TEMPLATE OPERATIONS
  seedPresets(projectId: string)
    → Promise<PresetSeedResult>
    POST /pipelines/{projectId}/seed-presets

  // ASSIGNMENT OPERATIONS
  getAssignment(projectId: string)
    → Promise<ProjectPipelineAssignment>
    GET /pipelines/{projectId}/assignment

  setAssignment(projectId: string, pipelineId: string)
    → Promise<ProjectPipelineAssignment>
    PUT /pipelines/{projectId}/assignment
    body: { pipeline_id: pipelineId }

  // LAUNCH OPERATIONS
  launch(projectId: string, data: PipelineIssueLaunchRequest)
    → Promise<WorkflowResult>
    POST /pipelines/{projectId}/launch
}

## 6. QUERY KEYS & REACT QUERY CONFIGURATION

FILE: /solune/frontend/src/hooks/usePipelineConfig.ts

QUERY KEY FACTORY:
  export const pipelineKeys = {
    all: ['pipelines'] as const,
    
    list: (projectId: string) 
      => [...pipelineKeys.all, 'list', projectId] as const,
    
    detail: (projectId: string, pipelineId: string)
      => [...pipelineKeys.all, 'detail', projectId, pipelineId] as const,
    
    assignment: (projectId: string)
      => [...pipelineKeys.all, 'assignment', projectId] as const,
  };

QUERY CONFIGURATION:
  • List pipelines:
    - queryKey: pipelineKeys.list(projectId ?? '')
    - queryFn: () => pipelinesApi.list(projectId!)
    - staleTime: STALE_TIME_SHORT
    - enabled: !!projectId

  • Get assignment:
    - queryKey: pipelineKeys.assignment(projectId ?? '')
    - queryFn: () => pipelinesApi.getAssignment(projectId!)
    - staleTime: STALE_TIME_SHORT
    - enabled: !!projectId

INVALIDATION PATTERNS:
  • After save/create: invalidateQueries({ queryKey: pipelineKeys.list(projectId) })
  • After assignment change: invalidateQueries({ queryKey: pipelineKeys.assignment(projectId) })
  • Optimistic updates: setQueryData(pipelineKeys.assignment(projectId), result)
